mcpdrop

The fastest way to add an MCP-powered AI copilot to any web app.

Usage
<!-- 1. Start the bridge -->
<!-- 2. Drop this in your app -->
<script src="@mcp-drop/core"></script>

<mcp-drop
  mcp-servers='[{"name":"bridge",
    "url":"http://localhost:3333"}]'
  persist-key
></mcp-drop>
